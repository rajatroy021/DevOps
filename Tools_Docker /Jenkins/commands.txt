# Pull Jenkins with Docker 

docker run -d -u root -v /var/jenkins/jenkins_home:/var/jenkins_home -v $(Which docker):/usr/bin/docker -v /var/run/docker.sock:/var/run/docker.sock -p 8080:8080 -p 50000:50000 --name jenkins-docker --priviliged jenkins/jenkins:lts


Start Docker as service on Ec2 ::

Sudo yum update 
Sudo yum -y install docker 

Sudo usermod a -G docker ec2-user
Sudo chmod 666 /var/run/docker.sock
sudo service docker start
